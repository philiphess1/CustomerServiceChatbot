Popup Chat

Copy and paste this code before the </body> tag on your website:

  • To learn more about embedding this code onto your Webflow site, click the link here.
  • To learn more about embedding this code onto your Wordpress site, click the link here.

            <img id="chatbot-button" src="http://127.0.0.1:5000//static/images/{{ settings['widget_icon'] }}" alt="Chat" style="width: 80px; height: 80px; position: fixed; bottom: 10px; right: 10px; z-index: 1000; cursor: pointer; transition: transform 0.3s ease-in-out;">
            <div id="p" style="display:{{popup_logic}};position:fixed;bottom:100px;right:15px;z-index:10000;background:#fff;border:1px solid #ccc;padding:10px;border-radius:5px;box-shadow:0 0 10px rgba(0,0,0,.1);">{{ settings['popup_message'] }}</div>
            <span id="c" style="display:none;position:fixed;bottom:128px;right:10px;z-index:10001;cursor:pointer;padding:2px 6px;border-radius:10px;background-color:#dadada;font-family: arial;color: #4e4e4e;">x</span>
            <script>
            var p=document.getElementById('p'),c=document.getElementById('c');
            p.onmouseover=c.onmouseover=function(){c.style.display='block'};
            p.onmouseout=function(){c.style.display='none'};
            c.onclick=function(){p.style.display=c.style.display='none'};
            </script>
            <iframe src="http://127.0.0.1:5000/17/11" id="chatbot-iframe" frameborder="0" scrolling="no" width="400" height="700" style="display: none; position:fixed; bottom:80px; right:0; z-index:9999;"></iframe>
            <script>
            var b=document.getElementById('chatbot-button'),i=document.getElementById('chatbot-iframe');
            b.onmouseover=function(){this.style.transform='scale(1.1)'};
            b.onmouseout=function(){this.style.transform='scale(1)'};
            b.onclick=()=>{
                i.style.display=i.style.display==='none'?'block':'none';
                if (i.style.display === 'block') {
                    p.style.display = 'none';
                }
                document[i.style.display==='block'?'addEventListener':'removeEventListener']('click',d);
            };
            d=e=>{if(e.target.id!=='chatbot-iframe'&&e.target.id!=='chatbot-button')i.style.display='none',document.removeEventListener('click',d)};
            </script>
